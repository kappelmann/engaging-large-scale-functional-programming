\section{I/O Mocking}

\begin{frame}{Motivation}

\begin{itemize}[<+->]
  \item Submissions (primarily) tested with QuickCheck
  \item I/O is an important part of the syllabus
\end{itemize}
\vfill
\visible<3->{
\begin{center}
\Large \alert{So how do we test I/O in Haskell?}
}
\end{center}

\end{frame}

\begin{frame}[fragile]{The Standard Way}
\begin{visibleenv}<3->
\begin{minted}[fontsize=\small]{haskell}
import qualified Prelude
import Prelude hiding (readFile, writeFile)

class Monad m => MonadFileSystem m where
  readFile :: FilePath -> m String
  writeFile :: FilePath -> String -> m ()
\end{minted}
\end{visibleenv}

\vspace{1.5em}

\begin{onlyenv}<1>
\begin{minted}[fontsize=\small]{haskell}
copyFile :: FilePath -> FilePath -> IO ()
copyFile = _
\end{minted}
\end{onlyenv}

\begin{onlyenv}<2-3>
\begin{minted}[fontsize=\small]{haskell}
copyFile :: MonadFileSystem m =>
            FilePath -> FilePath -> m ()
copyFile = _
\end{minted}
\end{onlyenv}

\begin{visibleenv}<4->
\begin{minted}[fontsize=\small]{haskell}
copyFile :: MonadFileSystem m =>
            FilePath -> FilePath -> m ()
copyFile source target = do
  content <- readFile source
  writeFile target content
\end{minted}
\end{visibleenv}
\end{frame}

\begin{frame}[fragile]{Multiple Instantiations}
\begin{minted}[fontsize=\small]{haskell}
instance MonadFileSystem IO where
  readFile = Prelude.readFile
  writeFile = Prelude.readFile
\end{minted}
\begin{visibleenv}<2>
\begin{minted}[fontsize=\small]{haskell}

data MockFileSystem =
  MockFileSystem (Map FilePath String)
instance MonadFileSystem (State MockFileSystem) where
  readFile = _
  writeFile = _
\end{minted}
\end{visibleenv}
\end{frame}

\begin{frame}[fragile]{The Problem}
What is the problem with
\begin{minted}[fontsize=\small]{haskell}
copyFile :: MonadFileSystem m =>
            FilePath -> FilePath -> m ()
copyFile = _
\end{minted}
\pause
\vfill
\begin{center}
  \alert{\Large Lack of transparency!}
\end{center}
\end{frame}

\begin{frame}{The Solution}
  \large
  Delay mocking to the compliation stage\\[1.5em]
  \pause
  \qquad by replacing the \texttt{IO} module with a mixin.
\end{frame}

\begin{frame}[fragile]{The Mixin}
\begin{minted}[fontsize=\small]{haskell}
data RealWord = RealWord {
  workDir :: FilePath,
  files :: Map File Text,
  handles :: Map Handle HandleData,
  user :: IO (),
  ...
}
\end{minted}
\begin{visibleenv}<2->
\begin{minted}[fontsize=\small]{haskell}

newtype IO a = IO { unwrapIO ::
  ExceptT IOException (PauseT (State RealWorld)) a }
\end{minted}
\end{visibleenv}
\end{frame}

\begin{frame}[fragile]{The \texttt{Pause} Monad}
\begin{minted}[fontsize=\small]{haskell}
class Monad m => MonadPause m where
  pause :: m ()
  stepPauseT :: m a -> m (Either (m a) a)
\end{minted}
\end{frame}

\begin{frame}[fragile]{An Example Interaction}
  \begin{columns}
    \begin{column}{0.5\textwidth}
\begin{minted}[fontsize=\small, highlightlines={1}]{haskell}
main = do
  x <- getLine
  putStrLn $ "Hello " ++ x
\end{minted}
    \end{column}
    \begin{column}{0.5\textwidth}
\begin{minted}[fontsize=\small, highlightlines={1}]{haskell}

  output <- hGetLine stdout
  when (_)
\end{minted}
    \end{column}
  \end{columns}
\end{frame}

\section{Check Your Proof}

\begin{frame}{CYP In A Nutshell}

\begin{itemize}[<+->]
  \item Operates on a strict, untyped subset of Haskell
  \item Automatically checks simple inductive proofs
  % \item Provides instant feedback
  \item Integrates with Tasty
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Background Theory}
\lstset{language=CYP}
  \begin{lstlisting}
data List a = [] | a : List a

[] ++ ys = ys
(x : xs) ++ ys = x : (xs ++ ys)

goal xs ++ (ys ++ zs) .=. (xs ++ ys) ++ zs
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{The \texttt{[]} Case}
  \begin{lstlisting}
Lemma: xs ++ (ys ++ zs) .=. (xs ++ ys) ++ zs
Proof by induction on List xs
  Case []
    To show: [] ++ (ys ++ zs) .=. ([] ++ ys) ++ zs
    Proof
                      [] ++ (ys ++ zs)
      (by def ++) .=. ys ++ zs
      (by def ++) .=. ([] ++ ys) ++ zs
    QED
  \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{The \texttt{Cons} Case}
  \begin{lstlisting}
  Case x : xs
    To show: (x : xs) ++ (ys ++ zs)
         .=. ((x : xs) ++ ys) ++ zs
    IH: xs ++ (ys ++ zs) .=. (xs ++ ys) ++ zs
    Proof
                        (x : xs) ++ (ys ++ zs)
      (by def ++)   .=. x : (xs ++ (ys ++ zs))
      (by IH)       .=. x : ((xs ++ ys) ++ zs)

                        ((x : xs) ++ ys) ++ zs
      (by def ++)   .=. (x : (xs ++ ys)) ++ zs
      (by def ++)   .=. x : ((xs ++ ys) ++ zs)
    QED
QED
  \end{lstlisting}
\end{frame}
